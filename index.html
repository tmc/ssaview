package main

func main() {
	a, b := 42, 43
	equal(a, b)

		a = 0
	} else {
		a = 1
	}
	fmt.Printf("a: %d | b: %d", a, b)
}


func equal(i, j int) bool {
	// needless closure
	inner := func() bool {
		return i == j
	}
	return inner()
}
